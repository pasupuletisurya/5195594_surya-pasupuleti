 def diagonalDifference(arr):
     x = 0
     y = 0
 
     for i in range(len(arr)):
        x += arr[i][i]
        y += arr[i][len(arr)-1-i]
     return(abs(x-y))

 def pangrams(s):
    alphabet={}
    for i in s.lower():
        if i.isalpha() and i in alphabet:
            alphabet[i]+=1
        elif i.isalpha() and i not in alphabet:
            alphabet[i]=0
    
    if len(alphabet)==26:
        return  "pangram"
    else:
        return "not pangram"
