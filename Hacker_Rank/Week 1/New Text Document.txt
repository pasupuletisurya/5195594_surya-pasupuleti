 def plusMinus(arr):

     negetiveCounter = 0

              positiveCounter += 1
         elif arr[i] < 0 :
              negetiveCounter +=1
         else:
              zeroCounter += 1

 def diagonalDifference(arr):
     x = 0
     y = 0
 
     for i in range(len(arr)):
        x += arr[i][i]
        y += arr[i][len(arr)-1-i]
     return(abs(x-y))
